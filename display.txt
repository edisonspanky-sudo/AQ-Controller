void updateLEDs() {
  static unsigned long lastBlink = 0;
  static bool blinkState = false;
  
  // Red LED - blinks during E-stop
  if (emergencyStop) {
    if (millis() - lastBlink > 500) {
      blinkState = !blinkState;
      buttonBox.digitalWrite(LED_RED, blinkState ? HIGH : LOW);
      lastBlink = millis();
      
      if (blinkState && !alarmSilenced) {
        tone(BUZZER_PIN, 2000, 100);
      }
    }
  } else {
    buttonBox.digitalWrite(LED_RED, LOW);
  }
  
  // Yellow LED - on during feed mode
  buttonBox.digitalWrite(LED_YELLOW, feedModeActive ? HIGH : LOW);
  
  // Blue LED - on when alarm active
  buttonBox.digitalWrite(LED_BLUE, (alarmSilenced || atoTimeoutAlarm || atoReservoirAlarm) ? HIGH : LOW);
  
  // Green LED - on during photo mode
  buttonBox.digitalWrite(LED_GREEN, photoModeActive ? HIGH : LOW);
}
